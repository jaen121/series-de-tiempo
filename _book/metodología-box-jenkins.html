7.2 Trasnformación de la serie en estacionaria

#transformar la serie
dif_dia <- diff(df_maiz_filtradots)

variacion_dia <- diff(df_maiz_filtradots) / lag(df_maiz_filtradots, -1)

# Gráfico de la serie
ts.plot(dif_dia, main = "Diferencia", col = "darkblue")

ts.plot(variacion_dia, main = "Variación", col = "darkred")
